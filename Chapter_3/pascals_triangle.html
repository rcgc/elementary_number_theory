// Will print Pascal's triangle

PROCEDURE pascals_triangle( limit ):
output = "1\n"
last = [1]
limit--

WHILE( limit >= 0 ):
add_at_beggining( 0 )
count = last.length()
last.append( 0 )
next = []

FOR( i = 0 UNTIL i < count STEP 1 ):
val = last[i] + last[i + 1]
next.append( val )
output += to_string( val ) + " "

output += "\n"
last = next
limit--
